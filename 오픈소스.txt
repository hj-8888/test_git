#!/bin/bash
echo "Select Command: "
select cmd in "현재 년도" "다음 년도" "이전 월" "현제 월" "다음 월" "종료"
do 
case $cmd in 
	"현재 년도") date +%Y;;
	"다음 년도") date -d '1 year' +%Y ;;
	"이전 월") date -d '1 month ago' +%m ;;
	"현제 월") date +%m;;
	"다음 월") date -d '1 month' +%m;;
	"종료") break;;
esac 
done
